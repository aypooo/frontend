let prevTarget = null

function openMenu(e){
    const target = e.target

    if(prevTarget !== null){
        prevTarget.nextElementSibling.classList.remove('open')
    }
    if(target.className === 'title'){
        console.dir(target)

        prevTarget = target
    }
}
